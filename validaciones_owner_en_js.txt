document.querySelector('#formOwner').addEventListener('submit', function(e) {
    e.preventDefault();  // Previene el envío hasta validar todo
    let isValid = true;  // Bandera para controlar si el formulario es válido

    const formData = new FormData(this);  // Obtiene los datos del formulario
    const name = formData.get('name').trim();  // Usa el nombre del campo
    const phone = formData.get('phone').trim();
    const email = formData.get('email').trim();

Longitud máxima: 50 caracteres.
Expresión regular: Debe admitir letras (mayúsculas y minúsculas) y espacios.


    // Validar que el nombre no esté vacío
    if (!checkRequired(name)) {
        document.querySelector('#nameError').innerText="El campo Nombre es requerido";
        isValid = false;
    }

    // Validar que el nombre no exceda los 50 caracteres
    if (name.length > 50) {
        document.querySelector('#nameError').innerText="El nombre no puede tener más de 50 caracteres";
        isValid = false;
    }

    // Validar que solo contenga letras y espacios
    const regexName = /^[A-Za-z\s]+$/;
    if (!regexName.test(name)) {
    document.querySelector('#nameError').innerText="El nombre solo puede contener letras y espacios";
        isValid = false;
    }

    // Validar que el teléfono no esté vacío
    if (!checkRequired(phone)) {
        document.querySelector('#phoneError').innerText="El campo Teléfono es requerido";
        isValid = false;
    }

    // Validar que el teléfono no exceda los 20 caracteres
    if (phone.length > 20) {
        document.querySelector('#phoneError').innerText="El teléfono no puede tener más de 20 caracteres;
        isValid = false;
    }

    // Validar que el formato del teléfono sea válido (ejemplo de formato de Argentina)
    const regexPhone = /^(?:\+54|54|0)?(11|[2-9]\d)\d{8}$/;
    if (!regexPhone.test(phone)) {
        document.querySelector('#phoneError').innerText="El formato del teléfono no es válido;
        isValid = false;
    }

    // Validar que el email no esté vacío
    if (!checkRequired(email)) {
        document.querySelector('#emailError').innerText="El campo Email es requerido.";
        isValid = false;
    }

    // Validar que el email no exceda los 80 caracteres
    if (email.length > 80) {
        document.querySelector('#emailError').innerText="El campo Email no puede tener más de 80 caracteres.";
        isValid = false;
    }

    // Validar la estructura del email
    const regexEmail = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!regexEmail.test(email)) {
        document.querySelector('#emailError').innerText="El campo Email no tiene una estructura válida.";
        isValid = false;
    }

    // Si todas las validaciones pasan, enviamos el formulario
    if (isValid) {
        this.submit();
    }
});

// Función para validar campos requeridos
function checkRequired(value) {
    return value !== '';
}
